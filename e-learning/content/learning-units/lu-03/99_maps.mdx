## BWC-Treaty States

import treaties from "../../data/treaties.json"

<LayeredMap
  centerLat={30}
  centerLong={2}
  mapScale={1.3}
  legendPosition="bottom-left"
  title="BWC Participation"
  license="cc-by-4"
  description="World map showing countries who have ratified or signed the BWC."
  caption="Global participation in the Biological Weapons Convention (BWC), updated weekly."
  credit="Data: [UNODA Disarmament Treaties Database](https://treaties.unoda.org/t/bwc/participants), [Natural Earth](https://www.naturalearthdata.com/). Graphic: Max Kohler/PRIF"
>
  <CountryStatisticsLayer data={
    treaties
      .find(el => { return el.name === "bwc"}).participants
      .map(p => {
        const s = ["ratification", "accession", "succession"].includes(p.events[p.events.length - 1].type) ? "Ratified" : "Signed"
        return ({iso3: p.country, value: s})
    })}
    statisticsGroupName="Participation Status" colors={["var(--blue-50)", "var(--blue-30)"]} />
</LayeredMap>
